1.1 Apa itu PHP?

PHP (Hypertext Preprocessor) adalah bahasa pemrograman server-side yang digunakan untuk pengembangan web dinamis.

Karakteristik PHP

  • Dijalankan di server, bukan browser
  • Ekstensi file: .php
  • Dapat disisipkan dalam HTML

Sejarah Singkat

  • Dibuat oleh Rasmus Lerdorf (1994)
  • Versi terbaru: PHP 8.x
  • Open Source & Gratis

Contoh Kode PHP Sederhana

<?php
    // Ini adalah komentar
    echo "Halo, Dunia!";
    print "Saya belajar PHP";
?>

/**
 * Sanitasi input data dari form
 * @param array $data Input data dari $_POST
 * @return array Data yang sudah dibersihkan
 */
function sanitize_input($data) {
    $clean_data = [];
    
    // Definisi field yang diperbolehkan
    $allowed_fields = ['nama', 'email', 'kelas'];
    
    foreach ($allowed_fields as $field) {
        if (isset($data[$field])) {
            // Hapus spasi di awal/akhir dan escape HTML
            $clean_data[$field] = htmlspecialchars(
                trim($data[$field]), 
                ENT_QUOTES, 
                'UTF-8'
            );
        }
    }
    
    return $clean_data;
}

/**
 * Validasi data siswa
 * @param array $data Data siswa yang divalidasi
 * @return array List error messages
 */
function validate_siswa_data($data) {
    $errors = [];
    
    // Validasi nama: wajib diisi, minimal 3 karakter
    if (empty($data['nama'])) {
        $errors[] = "Nama wajib diisi";
    } elseif (strlen($data['nama']) < 3) {
        $errors[] = "Nama minimal 3 karakter";
    }
    
    // Validasi email: wajib diisi, format email, unique
    if (empty($data['email'])) {
        $errors[] = "Email wajib diisi";
    } elseif (!filter_var($data['email'], FILTER_VALIDATE_EMAIL)) {
        $errors[] = "Format email tidak valid";
    } elseif (is_email_exists($data['email'])) {
        $errors[] = "Email sudah terdaftar";
    }
    
    // Validasi kelas: wajib dipilih
    if (empty($data['kelas'])) {
        $errors[] = "Kelas wajib dipilih";
    }
    
    return $errors;
}

/**
 * Cek apakah email sudah ada di database
 * @param string $email Email yang akan dicek
 * @return bool True jika email sudah ada
 */
function is_email_exists($email) {
    global $koneksi;
    
    $sql = "SELECT id FROM siswa WHERE email = ?";
    $stmt = mysqli_prepare($koneksi, $sql);
    mysqli_stmt_bind_param($stmt, "s", $email);
    mysqli_stmt_execute($stmt);
    $result = mysqli_stmt_get_result($stmt);
    
    return mysqli_num_rows($result) > 0;
}

/**
 * Proses pendaftaran siswa ke database
 * @param array $data Data siswa yang akan disimpan
 * @return string Success message
 */
function process_siswa_registration($data) {
    global $koneksi;
    
    // Insert data ke database dengan prepared statement
    $sql = "INSERT INTO siswa (nama, email, kelas, jurusan, created_at) 
                  VALUES (?, ?, ?, 'RPL', NOW())";
    
    $stmt = mysqli_prepare($koneksi, $sql);
    mysqli_stmt_bind_param($stmt, "sss", 
                             $data['nama'], 
                             $data['email'], 
                             $data['kelas']);
    
    if (mysqli_stmt_execute($stmt)) {
        return "‚úÖ Data siswa berhasil disimpan!";
    } else {
        return "‚ùå Gagal menyimpan data: " . mysqli_error($koneksi);
    }
}

/**
 * Mengambil semua data siswa dari database
 * @return array Data siswa yang sudah diurutkan
 */
function get_all_siswa() {
    global $koneksi;
    
    // Query dengan JOIN untuk menampilkan jurusan lengkap
    $sql = "SELECT 
                s.id, 
                s.nama, 
                s.email, 
                s.kelas, 
                s.jurusan,
                s.created_at,
                DATE_FORMAT(s.created_at, '%d %M %Y %H:%i') as tanggal_daftar_formatted
                FROM siswa s 
                ORDER BY s.created_at DESC";
    
    $result = mysqli_query($koneksi, $sql);
    $siswa_list = [];
    
    if (mysqli_num_rows($result) > 0) {
        while ($row = mysqli_fetch_assoc($result)) {
            $siswa_list[] = $row;
        }
    }
    
    return $siswa_list;
}

/**
 * Format status jurusan dengan badge

 */
function format_jurusan_badge($jurusan) {
    $badge_colors = [
        'RPL'   => 'primary',
        'TKJ'   => 'success',
        'MM'    => 'info',
        'AK'    => 'warning'
    ];
    
    $color = isset($badge_colors[$jurusan]) 
               ? $badge_colors[$jurusan] 
               : 'secondary';
    
    return "<span class='badge bg-$color'>$jurusan</span>";
}
